20const FW_CFG_CTL: u16 = 0x510;
21const FW_CFG_DATA: u16 = 0x511;
22
23const _FW_CFG_ID: u16 = 0x01;
24const FW_CFG_FILE_DIR: u16 = 0x19;
25
26// Must be a power-of-2
27const KERNEL_REGION_SIZE: u64 = 16 * 1024 * 1024;
28const KERNEL_REGION_SIZE_MASK: u64 = !(KERNEL_REGION_SIZE - 1);
29
30const MAX_FW_CFG_FILES: u32 = 0x1000;
31

34#[non_exhaustive]
35#[derive(Debug)]
36pub struct FwCfg<'a> {
37    driver: &'a dyn IOPort,
38}
39
40#[derive(Clone, Copy, Debug)]
41pub enum FwCfgError {
42    // Could not find the appropriate file selector.
43    FileNotFound,
44    // Unexpected file size.
45    FileSize(u32),
46    // Could not find an appropriate kernel region for the SVSM.
47    KernelRegion,
48    /// The firmware provided too many files to the guest
49    TooManyFiles,
50}
51
52impl From<FwCfgError> for SvsmError {
53    fn from(err: FwCfgError) -> Self {
54        Self::FwCfg(err)
55    }
56}
57
58#[derive(Debug, Clone, Copy)]
59pub struct FwCfgFile {
60    size: u32,
61    selector: u16,
62}
63
64impl FwCfgFile {
65    pub fn size(&self) -> u32 {
66        self.size
67    }
68    pub fn selector(&self) -> u16 {
69        self.selector
70    }
71}
72
73impl<'a> FwCfg<'a> {
74    pub fn new(driver: &'a dyn IOPort) -> Self {
75        FwCfg { driver }
76    }
77
78    pub fn select(&self, cfg: u16) {
79        self.driver.outw(FW_CFG_CTL, cfg);
80    }
81
82    pub fn read_bytes(&self, out: &mut [u8]) {
83        for byte in out.iter_mut() {
84            *byte = self.driver.inb(FW_CFG_DATA);
85        }
86    }
87
88    pub fn read_le<T>(&self) -> T
89    where
90        T: core::ops::Shl<usize, Output = T> + core::ops::BitOr<T, Output = T> + From<u8>,
91    {
92        let mut val = T::from(0u8);
93        let io = &self.driver;
94
95        for i in 0..size_of::<T>() {
96            val = (T::from(io.inb(FW_CFG_DATA)) << (i * 8)) | val;
97        }
98        val
99    }
100
101    pub fn read_be<T>(&self) -> T
102    where
103        T: core::ops::Shl<usize, Output = T> + core::ops::BitOr<T, Output = T> + From<u8>,
104    {
105        let mut val = T::from(0u8);
106        let io = &self.driver;
107
108        for _ in 0..size_of::<T>() {
109            val = (val << 8) | T::from(io.inb(FW_CFG_DATA));
110        }
111        val
112    }
113
114    pub fn read_u8(&self) -> u8 {
115        self.driver.inb(FW_CFG_DATA)
116    }
117
118    pub fn file_selector(&self, name: &str) -> Result<FwCfgFile, SvsmError> {
119        self.select(FW_CFG_FILE_DIR);
120        let n: u32 = self.read_be();
121
122        if n > MAX_FW_CFG_FILES {
123            return Err(SvsmError::FwCfg(FwCfgError::TooManyFiles));
124        }
125
126        for _ in 0..n {
127            let size: u32 = self.read_be();
128            let selector: u16 = self.read_be();
129            let _unused: u16 = self.read_be();
130            let mut st = String::with_capacity(56);
131            let mut terminated = false;
132            for _ in 0..56 {
133                let c = self.read_u8();
134                if terminated || c == b'\0' {
135                    terminated = true;
136                } else {
137                    st.push(c.into());
138                }
139            }
140
141            if st == name {
142                return Ok(FwCfgFile { size, selector });
143            }
144        }
145
146        Err(SvsmError::FwCfg(FwCfgError::FileNotFound))
147    }
148
149    fn find_svsm_region(&self) -> Result<MemoryRegion<PhysAddr>, SvsmError> {
150        let file = self.file_selector("etc/sev/svsm")?;
151
152        if file.size != 16 {
153            return Err(SvsmError::FwCfg(FwCfgError::FileSize(file.size)));
154        }
155
156        self.select(file.selector);
157        Ok(self.read_memory_region())
158    }
159
160    fn read_memory_region(&self) -> MemoryRegion<PhysAddr> {
161        let start = PhysAddr::from(self.read_le::<u64>());
162        let size = self.read_le::<u64>();
163        let end = start.saturating_add(size as usize);
164
165        assert!(start <= max_phys_addr(), "{start:#018x} is out of range");
166        assert!(end <= max_phys_addr(), "{end:#018x} is out of range");
167
168        MemoryRegion::from_addresses(start, end)
169    }
170
171    pub fn get_memory_regions(&self) -> Result<Vec<MemoryRegion<PhysAddr>>, SvsmError> {
172        let mut regions = Vec::new();
173        let file = self.file_selector("etc/e820")?;
174        let entries = file.size / 20;
175
176        self.select(file.selector);
177
178        for _ in 0..entries {
179            let region = self.read_memory_region();
180            let t: u32 = self.read_le();
181
182            if t == 1 {
183                regions.push(region);
184            }
185        }
186
187        Ok(regions)
188    }
189
190    fn find_kernel_region_e820(&self) -> Result<MemoryRegion<PhysAddr>, SvsmError> {
191        let regions = self.get_memory_regions()?;
192        let kernel_region = regions
193            .iter()
194            .max_by_key(|region| region.start())
195            .ok_or(SvsmError::FwCfg(FwCfgError::KernelRegion))?;
196
197        let start = PhysAddr::from(
198            kernel_region
199                .end()
200                .bits()
201                .saturating_sub(KERNEL_REGION_SIZE as usize)
202                & KERNEL_REGION_SIZE_MASK as usize,
203        );
204
205        if start < kernel_region.start() {
206            return Err(SvsmError::FwCfg(FwCfgError::KernelRegion));
207        }
208
209        Ok(MemoryRegion::new(start, kernel_region.len()))
210    }
211
212    pub fn find_kernel_region(&self) -> Result<MemoryRegion<PhysAddr>, SvsmError> {
213        let kernel_region = self
214            .find_svsm_region()
215            .or_else(|_| self.find_kernel_region_e820())?;
216
217        // Make sure that the kernel region doesn't overlap with the loader.
218        if kernel_region.start() < PhysAddr::from(u64::from(STAGE2_START + STAGE2_MAXLEN)) {
219            return Err(SvsmError::FwCfg(FwCfgError::KernelRegion));
220        }
221
222        Ok(kernel_region)
223    }
224
225    // This needs to be &mut self to prevent iterator invalidation, where the caller
226    // could do fw_cfg.select() while iterating. Having a mutable reference prevents
227    // other references.
228    pub fn iter_flash_regions(&self) -> impl Iterator<Item = MemoryRegion<PhysAddr>> + '_ {
229        let num = match self.file_selector("etc/flash") {
230            Ok(file) => {
231                self.select(file.selector);
232                file.size as usize / 16
233            }
234            Err(_) => 0,
235        };
236
237        (0..num).map(|_| self.read_memory_region())
238    }
