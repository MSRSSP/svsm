106pub fn idt_init() {
107    // Switch #PF handler to the default one
108    idt_mut().set_entry(PF_VECTOR, IdtEntry::entry(asm_entry_pf));
109
110    // Set IST vectors
111    init_ist_vectors();
112
113    // SAFETY:
114    // Capture an address that can be used by assembly code to read the #HV
115    // doorbell page.  The address of each CPU's doorbell page may be
116    // different, but the address of the field in the PerCpu structure that
117    // holds the actual pointer is constant across all CPUs, so that is the
118    // pointer that is actually captured.  The address that is captured is
119    // stored as a usize instead of a typed value, because the declarations
120    // required for type safety here are cumbersome, and the assembly code
121    // that uses the value is not type safe in any case, so enforcing type
122    // safety on the pointer would offer no meaningful value.
123    unsafe {
124        HV_DOORBELL_ADDR = this_cpu().hv_doorbell_addr() as usize;
125    };
126}

140// Doube-Fault handler
141#[no_mangle]
142extern "C" fn ex_handler_double_fault(ctxt: &mut X86ExceptionContext) {
143    let cr2 = read_cr2();
144    let rip = ctxt.frame.rip;
145    let rsp = ctxt.frame.rsp;
146
147    if user_mode(ctxt) {
148        log::error!(
149            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x} - Terminating task",
150            rip,
151            rsp,
152            cr2
153        );
154        terminate();
155    } else {
156        panic!(
157            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x}",
158            rip, rsp, cr2
159        );
160    }
161}

198// Page-Fault handler
199#[no_mangle]
200extern "C" fn ex_handler_page_fault(ctxt: &mut X86ExceptionContext, vector: usize) {
201    let cr2 = read_cr2();
202    let rip = ctxt.frame.rip;
203    let err = ctxt.error_code;
204    let vaddr = VirtAddr::from(cr2);
205
206    if user_mode(ctxt) {
207        let kill_task: bool = if is_task_fault(vaddr) {
208            current_task()
209                .fault(vaddr, (err & PageFaultError::W.bits() as usize) != 0)
210                .is_err()
211        } else {
212            true
213        };
214
215        if kill_task {
216            log::error!("Unexpected user-mode page-fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x} - Terminating task",
217                    rip, cr2, err);
218            terminate();
219        }
220    } else if this_cpu()
221        .handle_pf(
222            VirtAddr::from(cr2),
223            (err & PageFaultError::W.bits() as usize) != 0,
224        )
225        .is_err()
226        && !handle_exception_table(ctxt)
227    {
228        handle_debug_exception(ctxt, vector);
229        panic!(
230            "Unhandled Page-Fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x}",
231            rip, cr2, err
232        );
233    }
234}

236// Control-Protection handler
237#[no_mangle]
238extern "C" fn ex_handler_control_protection(ctxt: &mut X86ExceptionContext, _vector: usize) {
239    // From AMD64 Architecture Programmer's Manual, Volume 2, 8.4.3
240    // Control-Protection Error Code:
241    /// A RET (near) instruction encountered a return address mismatch.
242    const NEAR_RET: usize = 1;
243    /// A RET (far) or IRET instruction encountered a return address mismatch.
244    const FAR_RET_IRET: usize = 2;
245    /// An RSTORSSP instruction encountered an invalid shadow stack restore
246    /// token.
247    const RSTORSSP: usize = 4;
248    /// A SETSSBSY instruction encountered an invalid supervisor shadow stack
249    /// token.
250    const SETSSBSY: usize = 5;
251
252    let rip = ctxt.frame.rip;
253    match ctxt.error_code & 0x7fff {
254        code @ (NEAR_RET | FAR_RET_IRET) => {
255            // Read the return address on the normal stack.
256            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.frame.rsp));
257            // SAFETY: `rsp` is a valid guest address filled by the CPU in the
258            // X86InterruptFrame
259            let ret = unsafe { ret_ptr.read() }.expect("Failed to read return address");
260
261            // Read the return address on the shadow stack.
262            let prev_rssp_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.ssp));
263            // SAFETY: `ssp` is a valid guest address filled by the CPU in the
264            // X86ExceptionContext
265            let prev_rssp = unsafe { prev_rssp_ptr.read() }
266                .expect("Failed to read address of previous shadow stack pointer");
267            // The offset to the return pointer is different for RET and IRET.
268            let offset = if code == NEAR_RET { 0 } else { 8 };
269            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(prev_rssp + offset));
270            let ret_on_ssp =
271                // SAFETY: `ssp` is a valid guest address filled by the CPU in the
272                // X86ExceptionContext
273                unsafe { ret_ptr.read() }.expect("Failed to read return address on shadow stack");
274
275            panic!("thread at {rip:#018x} tried to return to {ret:#x}, but return address on shadow stack was {ret_on_ssp:#x}!");
276        }
277        RSTORSSP => {
278            panic!("rstorssp instruction encountered an unexpected shadow stack restore token at RIP {rip:#018x}");
279        }
280        SETSSBSY => {
281            panic!("setssbsy instruction encountered an unexpected supervisor shadow stack token at RIP {rip:#018x}");
282        }
283        code => unreachable!("unexpected code for #CP exception: {code}"),
284    }
285}

287// Virtualization Exception handler
288#[no_mangle]
289extern "C" fn ex_handler_ve(ctxt: &mut X86ExceptionContext) {
290    let rip = ctxt.frame.rip;
291    let code = ctxt.error_code;
292
293    if let Err(err) = handle_virtualization_exception(ctxt) {
294        log::error!("#VE handling error: {:?}", err);
295        if user_mode(ctxt) {
296            log::error!("Failed to handle #VE from user-mode at RIP {:#018x} code: {:#018x} - Terminating task", rip, code);
297            terminate();
298        } else {
299            panic!(
300                "Failed to handle #VE from kernel-mode at RIP {:#018x} code: {:#018x}",
301                rip, code
302            );
303        }
304    }
305}

327// System Call SoftIRQ handler
328#[no_mangle]
329extern "C" fn ex_handler_system_call(
330    ctxt: &mut X86ExceptionContext,
331    vector: usize,
332    event_type: IdtEventType,
333) {
334    // Ensure that this vector was not invoked as a hardware interrupt vector.
335    if event_type.is_external_interrupt(vector) {
336        panic!("Syscall handler invoked as external interrupt!");
337    }
338
339    if !user_mode(ctxt) {
340        panic!("Syscall handler called from kernel mode!");
341    }
342
343    let Ok(input) = TryInto::<u64>::try_into(ctxt.regs.rax) else {
344        ctxt.regs.rax = !0;
345        return;
346    };
347
348    ctxt.regs.rax = match input {
349        // Class 0 SysCalls.
350        SYS_EXIT => sys_exit(ctxt.regs.rdi as u32),
351        SYS_EXEC => sys_exec(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
352        SYS_CLOSE => sys_close(ctxt.regs.rdi as u32),
353        // Class 1 SysCalls.
354        SYS_OPEN => sys_open(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
355        SYS_READ => sys_read(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
356        SYS_WRITE => sys_write(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
357        SYS_SEEK => sys_seek(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
358        SYS_UNLINK => sys_unlink(ctxt.regs.rdi),
359        SYS_TRUNCATE => sys_truncate(ctxt.regs.rdi as u32, ctxt.regs.rsi),
360        SYS_OPENDIR => sys_opendir(ctxt.regs.rdi),
361        SYS_READDIR => sys_readdir(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
362        SYS_MKDIR => sys_mkdir(ctxt.regs.rdi),
363        SYS_RMDIR => sys_rmdir(ctxt.regs.rdi),
364        // Class 3 SysCalls.
365        SYS_CAPABILITIES => sys_capabilities(ctxt.regs.rdi as u32),
366        _ => Err(SysCallError::EINVAL),
367    }
368    .map_or_else(|e| e as usize, |v| v as usize);
369}

373    let rip = ctx.frame.rip;
374    let err = ctx.error_code;

383#[no_mangle]
384pub extern "C" fn common_isr_handler_entry(vector: usize) {
385    // Since interrupt handlers execute with interrupts disabled, it is
386    // necessary to increment the per-CPU interrupt disable nesting state
387    // while the handler is running in case common code attempts to disable
388    // interrupts temporarily.  The fact that this interrupt was received
389    // means that the previous state must have had interrupts enabled.
390    let cpu = this_cpu();
391    cpu.irqs_push_nesting(true);
392
393    common_isr_handler(vector);
394
395    // Decrement the interrupt disable nesting count, but do not permit
396    // interrupts to be reenabled.  They will be reenabled during the IRET
397    // flow.
398    cpu.irqs_pop_nesting();
399}
400
401pub fn common_isr_handler(vector: usize) {
402    // Set TPR based on the vector being handled and reenable interrupts to
403    // permit delivery of higher priority interrupts.  Because this routine
404    // dispatches interrupts which should only be observable if interrupts
405    // are enabled, the IRQ nesting count must be zero at this point.
406    let previous_tpr = raw_get_tpr();
407    raw_set_tpr(tpr_from_vector(vector));
408
409    let cpu = this_cpu();
410    cpu.irqs_enable();
411
412    // Process the requested interrupt vector.
413    match vector {
414        IPI_VECTOR => this_cpu().handle_ipi_interrupt(),
415        _ => {
416            // Ignore all unrecognized interrupt vectors and treat them as
417            // spurious interrupts.
418        }
419    }
420
421    // Disable interrupts before restoring TPR.
422    cpu.irqs_disable();
423    raw_set_tpr(previous_tpr);
424
425    // Perform the EOI cycle after the interrupt processing state has been
426    // restored so that recurrent interrupts will not introduce recursion at
427    // this point.
428    apic_eoi();
429}
