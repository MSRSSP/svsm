23/// vTPM platform commands (SVSM spec, section 8.1 - SVSM_VTPM_QUERY)
24///
25/// The platform commmand values follow the values used by the
26/// Official TPM 2.0 Reference Implementation by Microsoft.
27///
28/// `tpm-20-ref/TPMCmd/Simulator/include/TpmTcpProtocol.h`
29#[repr(u32)]
30#[derive(PartialEq, Copy, Clone, Debug)]
31pub enum TpmPlatformCommand {
32    SendCommand = 8,
33}
34
35impl TryFrom<u32> for TpmPlatformCommand {
36    type Error = SvsmReqError;
37
38    fn try_from(value: u32) -> Result<Self, Self::Error> {
39        let cmd = match value {
40            x if x == TpmPlatformCommand::SendCommand as u32 => TpmPlatformCommand::SendCommand,
41            other => {
42                log::warn!("Failed to convert {} to a TPM platform command", other);
43                return Err(SvsmReqError::invalid_parameter());
44            }
45        };
46
47        Ok(cmd)
48    }
49}

51fn vtpm_platform_commands_supported_bitmap() -> u64 {
52    let mut bitmap: u64 = 0;
53    let vtpm = vtpm_get_locked();
54
55    for cmd in vtpm.get_supported_commands() {
56        bitmap |= 1u64 << *cmd as u32;
57    }
58
59    bitmap
60}
61
62fn is_vtpm_platform_command_supported(cmd: TpmPlatformCommand) -> bool {
63    let vtpm = vtpm_get_locked();
64    vtpm.get_supported_commands().iter().any(|x| *x == cmd)
65}

67const SEND_COMMAND_REQ_INBUF_SIZE: usize = PAGE_SIZE - 9;
68
69// vTPM protocol services (SVSM spec, table 14)
70const SVSM_VTPM_QUERY: u32 = 0;
71const SVSM_VTPM_COMMAND: u32 = 1;
72
73/// TPM_SEND_COMMAND request structure (SVSM spec, table 16)
74#[derive(Clone, Copy, Debug)]
75#[repr(C, packed)]
76struct TpmSendCommandRequest {
77    /// MSSIM platform command ID
78    command: u32,
79    /// Locality usage for the vTPM is not defined yet (must be zero)
80    locality: u8,
81    /// Size of the input buffer
82    inbuf_size: u32,
83    /// Input buffer that contains the TPM command
84    inbuf: [u8; SEND_COMMAND_REQ_INBUF_SIZE],
85}
86
87impl TpmSendCommandRequest {
88    // Take as slice and return a reference for Self
89    pub fn try_from_as_ref(buffer: &[u8]) -> Result<&Self, SvsmReqError> {
90        let buffer = buffer
91            .get(..size_of::<Self>())
92            .ok_or_else(SvsmReqError::invalid_parameter)?;
93
94        // SAFETY: TpmSendCommandRequest has no invalid representations, as it
95        // is comprised entirely of integer types. It is repr(packed), so its
96        // required alignment is simply 1. We have checked the size, so this
97        // is entirely safe.
98        let request = unsafe { &*buffer.as_ptr().cast::<Self>() };
99
100        Ok(request)
101    }
102
103    pub fn send(&self) -> Result<Vec<u8>, SvsmReqError> {
104        // TODO: Before implementing locality, we need to agree what it means
105        // to the platform
106        if self.locality != 0 {
107            return Err(SvsmReqError::invalid_parameter());
108        }
109
110        let mut length = self.inbuf_size as usize;
111
112        let tpm_cmd = self
113            .inbuf
114            .get(..length)
115            .ok_or_else(SvsmReqError::invalid_parameter)?;
116        let mut buffer: Vec<u8> = Vec::with_capacity(SEND_COMMAND_RESP_OUTBUF_SIZE);
117        buffer.extend_from_slice(tpm_cmd);
118
119        // The buffer slice must be large enough to hold the TPM command response
120        buffer.resize(SEND_COMMAND_RESP_OUTBUF_SIZE, 0);
121
122        let vtpm = vtpm_get_locked();
123        vtpm.send_tpm_command(buffer.as_mut_slice(), &mut length, self.locality)?;
124
125        if length > buffer.len() {
126            return Err(SvsmReqError::invalid_request());
127        }
128        buffer.truncate(length);
129
130        Ok(buffer)
131    }
132}
133
134const SEND_COMMAND_RESP_OUTBUF_SIZE: usize = PAGE_SIZE - 4;
135
136/// TPM_SEND_COMMAND response structure (SVSM spec, table 17)
137#[derive(Clone, Copy, Debug)]
138#[repr(C, packed)]
139struct TpmSendCommandResponse {
140    /// Size of the output buffer
141    outbuf_size: u32,
142    /// Output buffer that will hold the command response
143    outbuf: [u8; SEND_COMMAND_RESP_OUTBUF_SIZE],
144}
145
146impl TpmSendCommandResponse {
147    // Take as slice and return a &mut Self
148    pub fn try_from_as_mut_ref(buffer: &mut [u8]) -> Result<&mut Self, SvsmReqError> {
149        let buffer = buffer
150            .get_mut(..size_of::<Self>())
151            .ok_or_else(SvsmReqError::invalid_parameter)?;
152
153        // SAFETY: TpmSendCommandResponse has no invalid representations, as it
154        // is comprised entirely of integer types. It is repr(packed), so its
155        // required alignment is simply 1. We have checked the size, so this
156        // is entirely safe.
157        let response = unsafe { &mut *buffer.as_mut_ptr().cast::<Self>() };
158
159        Ok(response)
160    }
161
162    /// Write the response to the outbuf
163    ///
164    /// # Arguments
165    ///
166    /// * `response`: TPM_SEND_COMMAND response slice
167    pub fn set_outbuf(&mut self, response: &[u8]) -> Result<(), SvsmReqError> {
168        self.outbuf
169            .get_mut(..response.len())
170            .ok_or_else(SvsmReqError::invalid_request)?
171            .copy_from_slice(response);
172        self.outbuf_size = response.len() as u32;
173
174        Ok(())
175    }
176}

178fn vtpm_query_request(params: &mut RequestParams) -> Result<(), SvsmReqError> {
179    // Bitmap of the supported vTPM commands
180    params.rcx = vtpm_platform_commands_supported_bitmap();
181    // Supported vTPM features. Must-be-zero
182    params.rdx = 0;
183
184    Ok(())
185}

187/// Send a TpmSendCommandRequest to the vTPM
188///
189/// # Arguments
190///
191/// * `buffer`: Contains the TpmSendCommandRequest. It will also be
192///             used to store the TpmSendCommandResponse as a byte slice
193///
194/// # Returns
195///
196/// * `u32`: Number of bytes written back to `buffer` as part of
197///          the TpmSendCommandResponse
198fn tpm_send_command_request(buffer: &mut [u8]) -> Result<u32, SvsmReqError> {
199    let outbuf: Vec<u8> = {
200        let request = TpmSendCommandRequest::try_from_as_ref(buffer)?;
201        request.send()?
202    };
203    let response = TpmSendCommandResponse::try_from_as_mut_ref(buffer)?;
204    let _ = response.set_outbuf(outbuf.as_slice());
205
206    Ok(outbuf.len() as u32)
207}
208
209fn vtpm_command_request(params: &RequestParams) -> Result<(), SvsmReqError> {
210    let paddr = PhysAddr::from(params.rcx);
211
212    if paddr.is_null() {
213        return Err(SvsmReqError::invalid_parameter());
214    }
215    if !valid_phys_address(paddr) {
216        return Err(SvsmReqError::invalid_address());
217    }
218
219    // The vTPM buffer size is one page, but it not required to be page aligned.
220    let start = paddr.page_align();
221    let offset = paddr.page_offset();
222    let end = (paddr + PAGE_SIZE).page_align_up();
223
224    let guard = PerCPUPageMappingGuard::create(start, end, 0)?;
225    let vaddr = guard.virt_addr() + offset;
226
227    // vTPM common request/response structure (SVSM spec, table 15)
228    //
229    // First 4 bytes are used as input and output.
230    //     IN: platform command
231    //    OUT: platform command response size
232
233    // SAFETY: vaddr comes from a new mapped region.
234    let command = unsafe { GuestPtr::<u32>::new(vaddr).read()? };
235
236    let cmd = TpmPlatformCommand::try_from(command)?;
237
238    if !is_vtpm_platform_command_supported(cmd) {
239        return Err(SvsmReqError::unsupported_call());
240    }
241
242    let buffer = unsafe { from_raw_parts_mut(vaddr.as_mut_ptr::<u8>(), PAGE_SIZE) };
243
244    let response_size = match cmd {
245        TpmPlatformCommand::SendCommand => tpm_send_command_request(buffer)?,
246    };
247
248    // SAFETY: vaddr points to a new mapped region.
249    // if paddr + sizeof::<u32>() goes to the folowing page, it should
250    // not be a problem since the end of the requested region is
251    // (paddr + PAGE_SIZE), which requests another page. So
252    // write(response_size) can only happen on valid memory, mapped
253    // by PerCPUPageMappingGuard::create().
254    unsafe {
255        GuestPtr::<u32>::new(vaddr).write(response_size)?;
256    }
257
258    Ok(())
259}
260
261pub fn vtpm_protocol_request(request: u32, params: &mut RequestParams) -> Result<(), SvsmReqError> {
262    match request {
263        SVSM_VTPM_QUERY => vtpm_query_request(params),
264        SVSM_VTPM_COMMAND => vtpm_command_request(params),
265        _ => Err(SvsmReqError::unsupported_call()),
266    }
